43      private void luoKomponentit(Container container) {
44          
45          Vector listData = new Vector();
46          JList listbox = new JList(listData);
47  //        listbox.addListSelectionListener(this);
48          JTextField syote = new JTextField();
49          
50          JPanel menu = new JPanel(new GridLayout(1, 3));
51          menu.add(listbox);
52          menu.add(syote);
53          container.add(menu);
54          
55          JPanel valikko = luovalikko(listbox, syote, listData);
56          container.add(valikko, BorderLayout.SOUTH);
57      }
58      
59      private JPanel luovalikko(JList vasen, JTextField oikea, Vector listData) {
60          JPanel panel = new JPanel(new GridLayout(1, 3));
61          JButton plus = new JButton("+");
62          JButton miinus = new JButton("-");
63          JButton nollaa = new JButton("Z");
64          Kuuntelija kuuntelija = new Kuuntelija(vasen, oikea, plus, miinus, nollaa, listData);
65          plus.addActionListener(kuuntelija);
66          miinus.addActionListener(kuuntelija);
67  //        nollaa.addActionListener(kuuntelija);
68          panel.add(plus);
69          panel.add(miinus);
70  //        nollaa.setEnabled(false);
71  //        panel.add(nollaa);
72          return panel;
73      }
74  
75      public JFrame getFrame() {
76          return frame;
77      }
