3   import drinkkikone.osat.Ainesosa;
4   import drinkkikone.osat.Drinkki;
5   import java.io.File;
6   import java.io.FileOutputStream;
7   import java.io.IOException;
8   import java.io.OutputStreamWriter;
9   
10  public class Tiedostoonkirjoittaja {
11      OutputStreamWriter kirjoittaja;
12  
13      public Tiedostoonkirjoittaja(boolean sailyykoVanha, String polku) throws IOException {
14          kirjoittaja = new OutputStreamWriter(new FileOutputStream(polku, sailyykoVanha),"UTF-8");
15      }
16  
17      public void lisaaTiedostoon(Drinkki drinkki) throws IOException {
18          kirjoittaja.write(drinkki.getNimi() + "#" + drinkki.getResepti() + "\n");
19          for (Ainesosa osa : drinkki.getAineet()) {
20              kirjoittaja.write(osa.getNimi() + "#" + osa.getMaara() + "\n");
21          }
22          kirjoittaja.write("\n");
23          kirjoittaja.close();
24      }
25  }
