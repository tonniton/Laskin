14  public class Tiedostonlukija {
15      private File tiedosto;
16      private Scanner lukija;
17      private Kirjanpito kirjanpito;
18  
19      /**
20       * fth
21       * @param kirjanpito
22       * @throws FileNotFoundException 
23       */
24      public Tiedostonlukija(Kirjanpito kirjanpito, String polku) throws FileNotFoundException {
25          this.tiedosto = new File(polku);
26          this.lukija = new Scanner(tiedosto, "UTF-8");
27          this.kirjanpito = kirjanpito;
28      }
29  
30      /**
31       * 
32       */
33      public void lueTiedosto() {
34          while (lukija.hasNextLine()) {
35              Drinkki drinkki = lisaaDrinkki();
36              lisaaAineet(drinkki);
37              kirjanpito.setDrinkki(drinkki);
38          }
39          lukija.close();
40      }
41  
42      private Drinkki lisaaDrinkki() {
43          String rivi = lukija.nextLine();
44          String[] sanat = rivi.split("#");
45          return new Drinkki(sanat[0], sanat[1]);
46      }
47  
48      private void lisaaAineet(Drinkki drinkki) {
49          String rivi = lukija.nextLine();
50          while (!rivi.isEmpty()) {
51              String[] sanat = rivi.split("#");
52              drinkki.setAine(new Ainesosa(sanat[0], sanat[1]));
53              if (!lukija.hasNextLine()) break;
54              rivi = lukija.nextLine();
55          } 
56      }
