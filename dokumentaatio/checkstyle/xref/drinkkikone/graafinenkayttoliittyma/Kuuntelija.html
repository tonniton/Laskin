6   package drinkkikone.graafinenkayttoliittyma;
7   
8   import java.awt.event.ActionEvent;
9   import java.awt.event.ActionListener;
10  import java.util.Vector;
11  import javax.swing.JButton;
12  import javax.swing.JList;
13  import javax.swing.JTextField;
14  
15  public class Kuuntelija implements ActionListener {
16  
17      private JList osat;
18      private JTextField lisaa;
19      private JButton plus;
20      private JButton miinus;
21      private JButton nollaa;
22      private Vector listData;
23  
24      public Kuuntelija(JList kentta1, JTextField kentta2, JButton lisaa, JButton miinus, JButton nollaa, Vector listData) {
25          this.osat = kentta1;
26          this.lisaa = kentta2;
27          this.plus = lisaa;
28          this.miinus = miinus;
29          this.nollaa = nollaa;
30          this.listData = listData;
31      }
32  
33      @Override
34      public void actionPerformed(ActionEvent ae) {
35          try {
36              String lisattava = lisaa.getText();
37              if (ae.getSource() == plus) {
38                  lisaa.setText("");
39                  listData.addElement(lisattava);
40                  osat.setListData(listData);
41              }
42              if (ae.getSource() == miinus) {
43                  int valittu = osat.getSelectedIndex();
44                  if (valittu >= 0) {
45                      listData.removeElementAt(valittu);
46                      osat.setListData(listData);
47                      if (valittu >= listData.size()) {
48                          valittu = listData.size() - 1;
49                      }
50                      osat.setSelectedIndex(valittu);
51                  }
52              }
53          } catch (Exception e) {
54  
55          }
56  
57      }
58  }
